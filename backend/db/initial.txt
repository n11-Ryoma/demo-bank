CREATE DATABASE ebank;


\c ebank;

-- ユーザ（認証情報）
CREATE TABLE users (
    id SERIAL PRIMARY KEY,
    username VARCHAR(255) UNIQUE NOT NULL,
    password VARCHAR(255) NOT NULL
);

-- ロール（権限）
CREATE TABLE roles (
    id SERIAL PRIMARY KEY,
    name VARCHAR(255) NOT NULL
);

-- ユーザ ↔ ロール
CREATE TABLE user_roles (
    user_id INTEGER REFERENCES users(id),
    role_id INTEGER REFERENCES roles(id)
);

-- 銀行口座
CREATE TABLE accounts (
    id SERIAL PRIMARY KEY,
    user_id INTEGER REFERENCES users(id),
    account_number VARCHAR(255) UNIQUE NOT NULL,
    balance BIGINT NOT NULL DEFAULT 0
);

-- 取引履歴（送金や入出金）
CREATE TABLE transactions (
    id SERIAL PRIMARY KEY,
    account_id INTEGER REFERENCES accounts(id),
    amount BIGINT NOT NULL,
    type VARCHAR(50) NOT NULL, -- "DEPOSIT", "WITHDRAW", "TRANSFER"
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- accounts テーブル作成
CREATE TABLE accounts (
    id              BIGSERIAL PRIMARY KEY,          -- 口座レコードID
    user_id         BIGINT NOT NULL,                -- users.id へのFK
    account_number  VARCHAR(20) NOT NULL,           -- 口座番号 (例: 0001-0000001)
    balance         BIGINT NOT NULL DEFAULT 0,      -- 残高（円）
    is_main         BOOLEAN NOT NULL DEFAULT FALSE, -- メイン口座フラグ
    created_at      TIMESTAMPTZ NOT NULL DEFAULT NOW(),
    updated_at      TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

-- users テーブルへの外部キー制約
ALTER TABLE accounts
  ADD CONSTRAINT fk_accounts_user
  FOREIGN KEY (user_id)
  REFERENCES users (id)
  ON DELETE CASCADE;

-- 口座番号はユニークにしておく
CREATE UNIQUE INDEX ux_accounts_account_number
  ON accounts (account_number);

-- 「ユーザごとにメイン口座は1つまで」にしたい場合の部分ユニークインデックス
CREATE UNIQUE INDEX ux_accounts_user_main
  ON accounts (user_id)
  WHERE is_main = TRUE;

CREATE TABLE transactions (
    id BIGSERIAL PRIMARY KEY,
    account_id      BIGINT NOT NULL,              -- この口座視点の取引
    type            VARCHAR(20) NOT NULL,        -- DEPOSIT / WITHDRAW / TRANSFER_IN / TRANSFER_OUT
    amount          BIGINT NOT NULL,             -- 絶対値（金額）。マイナスにはしない設計
    balance_after   BIGINT NOT NULL,             -- この取引後の残高（将来の履歴表示用）
    related_account_number VARCHAR(50),          -- 振込相手など
    description     VARCHAR(255),                -- メモ（ATM入金 etc）
    created_at      TIMESTAMPTZ NOT NULL DEFAULT CURRENT_TIMESTAMP
);
